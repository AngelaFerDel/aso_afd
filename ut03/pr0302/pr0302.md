# Unidad 3 - Práctica 2: Ejercicios con `case`.

## Ejercicio 1 - Menú de operaciones matemáticas.

```
    #!/bin/bash

    echo "--  MENÚ DE OPERACIONES  --"
    echo "1) Suma"
    echo "2) Resta"
    echo "3) Multiplicación"
    echo "4) División"
    read -p "Elige una opción [1-4]: " opcion

    read -p "Introduce el primer número: " num1
    read -p "Introduce el segundo número: " num2

    case $opcion in
    1) resultado=$((num1 + num2)) ;;
    2) resultado=$((num1 - num2)) ;;
    3) resultado=$((num1 * num2)) ;;
    4)
        if [ $num2 -ne 0 ]; then
            resultado=$((num1 / num2))
        else
            echo "Error. No se puede dividir entre 0."
            exit 1
        fi
        ;;
    *) echo "Opción no válida"; exit 1 ;;
    esac
    echo "Resultado: $resultado"
```

## Ejercicio 2 - Identificación de extensión de archivo.

```
    #!/bin/bash

    read -p "Introduce el nombre del archivo: " archivo

    case "$archivo" in
    *.txt) echo "Es un archivo de texto" ;;
    *.zip) echo "Es un archivo comprimido ZIP" ;;
    *.tar.gz) echo "Es un archivo comprimido TAR.GZ" ;;
    *) echo "Tipo de archivo desconocido" ;;
    esac
```

## Ejercicio 3 - Conversor de unidades.

```
    #!/bin/bash

    echo "=== Conversor de unidades ==="
    echo "1) Metros a kilómetros"
    echo "2) Kilómetros a metros"
    echo "3) Pies a metros"
    echo "4) Metros a pies"
    read -p "Selecciona una opción [1-4]: " opcion
    read -p "Introduce el valor a convertir: " valor

    case $opcion in
    1) resultado=$((valor / 1000)) ;;
    2) resultado=$((valor * 1000)) ;;
    3) resultado=$((valor * 3048 / 10000)) ;;
    4) resultado=$((valor * 10000 / 3048)) ;;
    *) echo "Opción no válida"; exit 1 ;;
    esac

    echo "Resultado aproximado: $resultado"
```

## Ejercicio 4 - Menú de configuración del sistema.

```
    #!/bin/bash

    echo "--  CONFIGURACIÓN DEL SISTEMA  --"
    echo "1) Apagar"
    echo "2) Reiniciar"
    echo "3) Cerrar sesión"
    read -p "Selecciona una opción [1-3]: " opcion

    case $opcion in
    1) sudo shutdown now ;;
    2) sudo reboot ;;
    3) pkill -KILL -u $USER ;;
    *) echo "Opción no válida"; exit 1 ;;
    esac
```

## Ejercicio 8 - Control de servicios en Linux.

```
    #!/bin/bash

    read -p "Introduce el nombre del servicio" servicio

    echo "--  CONTROL DE SERVICIOS  --"
    echo "1) Iniciar servicio"
    echo "2) Detener servicio"
    echo "3) Reiniciar servicio"
    read -p "Selecciona una opción [1-3]: " opcion

    case $opcion in
    1) sudo systemctl start $servicio ;;
    2) sudo systemctl stop $servicio ;;
    3) sudo systemctl restart $servicio ;;
    *) echo "Opción no válida"; exit 1 ;;
    esac
```
